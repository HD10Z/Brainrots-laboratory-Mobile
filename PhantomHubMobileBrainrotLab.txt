local fileName = "PhantomHub_" .. tostring(Player.UserId) .. ".json"
local savedSettings = {}

if type(isfile) == "function" and isfile(fileName) then
    local success, data = pcall(function()
        return HttpService:JSONDecode(readfile(fileName))
    end)
    if success and type(data) == "table" then
        savedSettings = data
    end
end

local function saveSettings()
    if type(writefile) == "function" then
        pcall(function()
            writefile(fileName, HttpService:JSONEncode(savedSettings))
        end)
    end
end

local optionStates = {}

local function createOptionState(name, initialValue)
    if savedSettings[name] ~= nil then
        optionStates[name] = savedSettings[name]
    else
        optionStates[name] = initialValue
    end
    return function() return optionStates[name] end
end

local function setOptionState(name, value)
    optionStates[name] = value
    savedSettings[name] = value
    saveSettings()
end

local inputDebounce = {}

local function createDebounce(key, duration)
    if inputDebounce[key] then return false end
    inputDebounce[key] = true
    local inputType = UserInputService:GetLastInputType()
    local adjustedDuration = duration or (inputType == Enum.UserInputType.Mouse or inputType == Enum.UserInputType.MouseButton1 or inputType == Enum.UserInputType.MouseButton2) and 0.05 or 0.03
    task.delay(adjustedDuration, function() inputDebounce[key] = nil end)
    return true
end

    if createDebounce(name, 0.05) then
        optionStates[name] = not optionStates[name]
        setOptionState(name, optionStates[name])

        Circle:TweenPosition(optionStates[name] and UDim2.new(1, -21, 0.5, -9) or UDim2.new(0, 3, 0.5, -9), "Out", "Quad", 0.2, true)

            task.spawn(function() callback(optionStates[name]) end)
        end
    end
end

    if currentActive and callback then
        task.spawn(function() callback(currentActive) end)
    end

    table.insert(options[tab], Container)
    return function() return optionStates[name] end, Container, SwitchBG

    SliderBG.BackgroundColor3 = Color3.fromRGB(40, 40, 40)
    SliderBG.Text = ""
    SliderBG.Parent = Container
    SliderBG.AutoButtonColor = false
    Instance.new("UICorner", SliderBG).CornerRadius = UDim.new(1, 0)

    local SliderFill = Instance.new("Frame")
    local startScale = (currentVal - min) / (max - min)
    SliderFill.Size = UDim2.new(startScale, 0, 1, 0)
    SliderFill.BackgroundColor3 = Color3.fromRGB(150, 150, 150)
    SliderFill.Parent = SliderBG
    Instance.new("UICorner", SliderFill).CornerRadius = UDim.new(1, 0)

    local Circle = Instance.new("Frame")
    Circle.Size = UDim2.new(0, 14, 0, 14)
    Circle.Position = UDim2.new(1, -7, 0.5, -7)
    Circle.BackgroundColor3 = Color3.fromRGB(255, 255, 255)
    Circle.Parent = SliderFill
    Instance.new("UICorner", Circle).CornerRadius = UDim.new(1, 0)

    if callback then
        task.spawn(function() callback(currentVal) end)
    end

    local dragging = false
    local function updateSlider(input)
        local relativeX = math.clamp(input.Position.X - SliderBG.AbsolutePosition.X, 0, SliderBG.AbsoluteSize.X)
        local scale = relativeX / SliderBG.AbsoluteSize.X
        SliderFill.Size = UDim2.new(scale, 0, 1, 0)
        local value = math.floor(min + ((max - min) * scale))
        Label.Text = name .. ": " .. tostring(value)

        optionStates[name] = value
        setOptionState(name, value)

        if callback then callback(value) end
    end

    SliderBG.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        dragging = true
        updateSlider(input)
    end
end)
UserInputService.InputChanged:Connect(function(input)
    if dragging then
        if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Mouse or input.UserInputType == Enum.UserInputType.Touch then
            updateSlider(input)
        end
    end
end)
UserInputService.InputEnded:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        dragging = false
    end
end)

    table.insert(options[tab], Container)
end

local originalSettings = {}
local lowModeConn = nil

local function applyAgressiveLow(v)
    if v:IsA("BasePart") then
        if not originalSettings[v] then
            originalSettings[v] = {v.Material, v.Reflectance, v.CastShadow}
        end
        v.Material = Enum.Material.SmoothPlastic
        v.Reflectance = 0
        v.CastShadow = false
    elseif v:IsA("Decal") or v:IsA("Texture") then
        if not originalSettings[v] then originalSettings[v] = v.Transparency end
        v.Transparency = 1
    elseif v:IsA("ParticleEmitter") or v:IsA("Trail") or v:IsA("Beam") or v:IsA("Fire") or v:IsA("Smoke") or v:IsA("Sparkles") then
        if not originalSettings[v] then originalSettings[v] = v.Enabled end
        v.Enabled = false
    elseif v:IsA("PostEffect") or v:IsA("BloomEffect") or v:IsA("BlurEffect") or v:IsA("SunRaysEffect") or v:IsA("ColorCorrectionEffect") or v:IsA("Atmosphere") then
        if not originalSettings[v] then originalSettings[v] = v.Enabled end
        v.Enabled = false
    end
end

local isLowMode = createOption("Low Mode", false, "Main", function(active)
    if active then
        originalSettings["Lighting"] = {game.Lighting.GlobalShadows, game.Lighting.FogEnd, game.Lighting.Brightness}
        game.Lighting.GlobalShadows = false
        game.Lighting.Brightness = 2
        game.Lighting.FogEnd = 100000

        for _, v in pairs(workspace:GetDescendants()) do applyAgressiveLow(v) end
        for _, v in pairs(game.Lighting:GetChildren()) do applyAgressiveLow(v) end

        lowModeConn = workspace.DescendantAdded:Connect(function(child)
            task.wait()
            applyAgressiveLow(child)
        end)
    else
        if lowModeConn then lowModeConn:Disconnect() lowModeConn = nil end

        if originalSettings["Lighting"] then
            game.Lighting.GlobalShadows = originalSettings["Lighting"][1]
            game.Lighting.FogEnd = originalSettings["Lighting"][2]
            game.Lighting.Brightness = originalSettings["Lighting"][3]
        end

        for obj, data in pairs(originalSettings) do
            if typeof(obj) == "Instance" and obj.Parent then
                if obj:IsA("BasePart") then
                    obj.Material = data[1]
                    obj.Reflectance = data[2]

                    obj.Enabled = data
                end
            end
        end
        originalSettings = {}
    end
end)

            end
        end

    if not myBaseName or not telePart then return end

    if active then
        local num = tonumber(string.match(myBaseName, "%d+")) or 1
        local offsetZ = (num <= 4) and 1500 or -1500

        local targetCFrame = telePart.CFrame * CFrame.new(0, 0, offsetZ)

local function parseEarnValue(text)
    if not text then return 0 end
    local clean = text:gsub("[%$%/s%, ]", "")
    local suffix = clean:sub(-1):upper()
    local number = tonumber(clean:sub(1, -2)) or tonumber(clean)

    if not number then return 0 end
    if suffix == "K" then return number * 1000
    elseif suffix == "M" then return number * 1000000
    elseif suffix == "B" then return number * 1000000000
    elseif suffix == "T" then return number * 1000000000000
    end
    return tonumber(clean) or 0
end

local function formatValue(val)
    if val >= 1e12 then return string.format("%.1fT", val/1e12)
    elseif val >= 1e9 then return string.format("%.1fB", val/1e9)
    elseif val >= 1e6 then return string.format("%.1fM", val/1e6)
    elseif val >= 1e3 then return string.format("%.1fK", val/1e3)
    end
    return tostring(val)
end

RunService.Heartbeat:Connect(function()
    local char = Player.Character
    if char and char:FindFirstChild("Humanoid") then
        char.Humanoid.WalkSpeed = isSpeedEnabled() and customSpeedVal or 32
    end

    if isFlyActive and bg and bv then
        local cam = workspace.CurrentCamera
        local char = Player.Character
        if char and char:FindFirstChild("HumanoidRootPart") then
            local hum = char:FindFirstChildOfClass("Humanoid")
            if hum then
                local moveDir = hum.MoveDirection
                if moveDir.Magnitude > 0 then
                    local camDir = cam.CFrame.LookVector
                    local camRight = cam.CFrame.RightVector
                    
                    local flatLook = Vector3.new(camDir.X, 0, camDir.Z)
                    if flatLook.Magnitude > 0.001 then
                        flatLook = flatLook.Unit
                    else
                        flatLook = Vector3.new(cam.CFrame.UpVector.X, 0, cam.CFrame.UpVector.Z).Unit * -math.sign(camDir.Y)
                    end
                    
                    local flatRight = Vector3.new(camRight.X, 0, camRight.Z)
                    if flatRight.Magnitude > 0.001 then
                        flatRight = flatRight.Unit
                    else
                        flatRight = Vector3.new(1, 0, 0)
                    end
                    
                    local forwardInput = moveDir:Dot(flatLook)
                    local rightInput = moveDir:Dot(flatRight)
                    
                    bv.Velocity = (camDir * forwardInput + camRight * rightInput) * flySpeed
                else
                    bv.Velocity = Vector3.new(0, 0, 0)
                end
                bg.CFrame = cam.CFrame
            end
        end
    end

    if noclipEnabled then
        local char = Player.Character
        if char then
            for _, part in pairs(char:GetDescendants()) do
                if part:IsA("BasePart") then
                    part.CanCollide = false
                end
            end
        end
    end
end)

local function setVisibility(visible)
    local char = Player.Character
    if not char then return end
    local transparency = visible and 0 or 1
    for _, v in pairs(char:GetDescendants()) do
        if v:IsA("BasePart") or v:IsA("Decal") then
            if v.Name ~= "HumanoidRootPart" then v.Transparency = transparency end
        elseif v:IsA("BillboardGui") or v:IsA("SurfaceGui") then
            v.Enabled = visible
        end
    end
    isHidden = not visible
end

local function checkIndex(name, mutation)
    if MainModule and type(MainModule) == "table" and MainModule.Data and MainModule.Data.Index and MainModule.Data.Index[mutation] then
        for internalName, _ in pairs(MainModule.Data.Index[mutation]) do
            if internalName == name or (MainModule.Shared and MainModule.Shared.Brainrots and MainModule.Shared.Brainrots[internalName] and MainModule.Shared.Brainrots[internalName].Name == name) then
                return true
            end
        end
    end
    return false
end

RunService.Heartbeat:Connect(function()
    if not brainrotsFolder then return end

    local autoEpic = isAutoEpic()
    local autoLegendary = isAutoLegendary()
    local autoMystic = isAutoMystic()
    local autoSecret = isAutoSecret()
    local autoStellar = isAutoStellar()
    local autoIdx = isAutoIndex()
    local autoSpec = isAutoSpecific()
    local rActive = isRebirthAutoActive and currentRebirthSelect > 0
    local smartMode = isSmartTeleport()

    if not (autoEpic or autoLegendary or autoMystic or autoSecret or autoStellar or rActive or autoIdx or autoSpec) then
        targetHead = nil
        allValidHeads = {}
        currentTargetIndex = 1
        if isHidden then setVisibility(true) end
        return
    end

    local char = Player.Character
    local root = char and char:FindFirstChild("HumanoidRootPart")
    if not root then return end

    allValidHeads = {}

    for _, model in pairs(brainrotsFolder:GetChildren()) do
        if model:IsA("Model") then
            local ui = model:FindFirstChild("Brainrot_UI")
            local frame = ui and ui:FindFirstChild("Frame")
            local rarityLbl = frame and frame:FindFirstChild("Rarity")
            local titleLbl = frame and frame:FindFirstChild("Title")
            local typeLbl = frame and frame:FindFirstChild("Type")
            local head = model:FindFirstChild("Head")

            if head and frame then
                local rarityText = rarityLbl and string.lower(rarityLbl.Text) or ""
                local titleText = titleLbl and titleLbl.Text or ""
                local shouldAdd = false

                local currentMut = "Normal"
                local tText = typeLbl and typeLbl.Text or ""
                if tText:find("Gold") then currentMut = "Gold"
                elseif tText:find("Diamond") then currentMut = "Diamond"
                elseif tText:find("Rainbow") then currentMut = "Rainbow"
                elseif tText:find("Lava") then currentMut = "Lava" end

                if autoSpec and specificTargets[titleText] and specificTargets[titleText].Active and specificTargets[titleText].Muts[currentMut] then
                    shouldAdd = true
                elseif autoStellar and (rarityText:find("stellar") or stellarNames[titleText]) then
                    shouldAdd = true
                elseif autoSecret and (rarityText:find("secret") or secretNames[titleText]) then
                    if titleText == "Lucky Block" and not optionStates["AutoSecretLuckyBlocks"] then
                        shouldAdd = false
                    else
                        shouldAdd = true
                    end
                elseif autoMystic and (rarityText:find("mythic") or mythicNames[titleText]) then
                    shouldAdd = true
                elseif autoLegendary and (rarityText:find("legendary") or legendaryNames[titleText]) then
                    shouldAdd = true
                elseif autoEpic and (rarityText:find("epic") or epicNames[titleText]) then
                    shouldAdd = true
                elseif rActive and rebirthTargets[currentRebirthSelect] and rebirthTargets[currentRebirthSelect][titleText] then
                    shouldAdd = true
                elseif autoIdx and mutSelected[currentMut] and titleText ~= "Lucky Block" and not checkIndex(titleText, currentMut) then
                    shouldAdd = true
                end

                if shouldAdd then
                    table.insert(allValidHeads, head)
                end
            end
        end
    end

    if #allValidHeads > 0 then
        if smartMode then
            if currentTargetIndex > #allValidHeads then
                currentTargetIndex = 1
            end
            targetHead = allValidHeads[currentTargetIndex]
            currentTargetIndex = currentTargetIndex + 1
        else
            targetHead = allValidHeads[1]
        end

        if targetHead and targetHead.Parent then
            root.CFrame = targetHead.CFrame
            if not isHidden then setVisibility(false) end
        end
    else
        targetHead = nil
        if isHidden then setVisibility(true) end
    end
end)
